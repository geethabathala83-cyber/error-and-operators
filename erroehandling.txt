#include <stdio.h>

int main() {
    int a, b;
    float result;

    printf("Enter numerator: ");
    scanf("%d", &a);

    printf("Enter denominator: ");
    scanf("%d", &b);

    // Error handling: check for division by zero
    if (b == 0) {
        printf("Error: Division by zero is not allowed.\n");
    } else {
        result = (float)a / b;
        printf("Result: %.2f\n", result);
    }

    return 0;
}
